```csharp
using Fiap.FCGames.Catalogo.Domain.Aggregates.AggregateBiblioteca;
using Fiap.FCGames.Catalogo.Infra.DataProvider.Contexto;
using Fiap.FCGames.Catalogo.Infra.DataProvider.Interface;
using Fiap.FCGames.Catalogo.Infra.DataProvider.Repositories.Shared;
using Microsoft.EntityFrameworkCore;

namespace Fiap.FCGames.Catalogo.Infra.DataProvider.Repositories;

public class BibliotecaRepository : GenericRepository<Biblioteca>, IBibliotecaRepository
{
    private readonly FcGamesContexto _context;

    public BibliotecaRepository(FcGamesContexto context) : base(context)
    {
        _context = context;
    }

    public async Task<Biblioteca?> ObterPorUsuarioIdAsync(Guid usuarioId)
        => await Get(b => b.UsuarioId == usuarioId).AsNoTracking().FirstOrDefaultAsync();

    public async Task<Biblioteca?> ObterComItensPorUsuarioIdAsync(Guid usuarioId)
        => await _context.Bibliotecas
            .Include(b => b.Itens)
            .AsNoTracking()
            .FirstOrDefaultAsync(b => b.UsuarioId == usuarioId);

    public async Task<bool> UsuarioPossuiJogoAsync(Guid usuarioId, Guid jogoId)
    {
        var biblioteca = await ObterPorUsuarioIdAsync(usuarioId);
        if (biblioteca is null) return false;

        return await _context.ItensBiblioteca
            .AsNoTracking()
            .AnyAsync(i => i.BibliotecaId == biblioteca.Id && i.JogoId == jogoId);
    }

    public void Adicionar(Biblioteca biblioteca) => Create(biblioteca);

    public void AdicionarItem(ItemBiblioteca item) => _context.ItensBiblioteca.Add(item);
}
```

    public static IServiceCollection AddMassTransitRabbitMq(

public class RealizarCompraCommandHandler : IRequestHandler<RealizarCompraCommand, RealizarCompraResponse>

    private readonly IUnitOfWork _uow;

    private readonly ILogger<RealizarCompraCommandHandler> _logger;

        IUnitOfWork uow,

        _uow = uow;
        _publishEndpoint = publishEndpoint;
        _logger = logger;

    {

            ?? throw new NotFoundException($"Jogo {request.JogoId} não encontrado.");

            Preco = jogo.Preco,
            Status = StatusPedido.Pendente,
            CriadoEm = DateTime.UtcNow
        };

        _uow.PedidoRepository.Adicionar(pedido);
        await _uow.CommitAsync(cancellationToken);

        // Publica evento APÓS commit — broker failure não corrompe o pedido
        try
        {

                UsuarioId: pedido.UsuarioId,

                RealizadoEmUtc: pedido.CriadoEm,
                CorrelationId: Guid.NewGuid()

        {
            _logger.LogError(ex, "Falha ao publicar PedidoRealizadoEvento para pedido {PedidoId}. Pedido persistido com sucesso.", pedido.Id);
        }

        return new RealizarCompraResponse(pedido.Id, pedido.Status.ToString());

## Task 13 — Consumer: PagamentoProcessadoEvento (FASE 4 complete)

**Purpose:** When a payment is processed, update the Pedido status and (if Aprovado) add ItemBiblioteca. Idempotent by PedidoId.

- [ ] **Step 1: Create PagamentoProcessadoConsumer**

Create `Fiap.FCGames.Catalogo.Api\Consumers\PagamentoProcessadoConsumer.cs`:

```csharp
// NOTE: Consumers live in the Api project for now.
// Extract to .Worker project if the team decides to separate concerns.
using FCGames.IntegrationEvents;
using Fiap.FCGames.Catalogo.Domain.Aggregates.AggregateBiblioteca;
using Fiap.FCGames.Catalogo.Domain.Enums;
using Fiap.FCGames.Catalogo.Infra.DataProvider.Interface;
using MassTransit;
using Microsoft.Extensions.Logging;

namespace Fiap.FCGames.Catalogo.Api.Consumers;

public class PagamentoProcessadoConsumer : IConsumer<PagamentoProcessadoEvento>
{
    private readonly IUnitOfWork _uow;
    private readonly ILogger<PagamentoProcessadoConsumer> _logger;

    public PagamentoProcessadoConsumer(IUnitOfWork uow, ILogger<PagamentoProcessadoConsumer> logger)
    {
        _uow = uow;
        _logger = logger;
    }

    public async Task Consume(ConsumeContext<PagamentoProcessadoEvento> context)
    {
        var evento = context.Message;

        _logger.LogInformation(
            "PagamentoProcessadoEvento recebido. PedidoId={PedidoId} Status={Status} CorrelationId={CorrelationId}",
            evento.PedidoId, evento.Status, evento.CorrelationId);

        // Idempotência: carrega o pedido atual e verifica se já foi processado
        var pedido = await _uow.PedidoRepository.ObterPorIdAsync(evento.PedidoId);
        if (pedido is null)
        {
            _logger.LogWarning("Pedido {PedidoId} não encontrado. Evento ignorado.", evento.PedidoId);
            return;
        }

        if (pedido.Status != StatusPedido.Pendente)
        {
            _logger.LogInformation("Pedido {PedidoId} já processado (Status={Status}). Ignorando.", pedido.Id, pedido.Status);
            return;
        }

        if (evento.Status == "Aprovado")
        {
            pedido.Status = StatusPedido.Aprovado;
            _uow.PedidoRepository.Atualizar(pedido);

            // Adiciona jogo à biblioteca (cria biblioteca se não existir — usuário pode não ter cadastrado ainda)
            var biblioteca = await _uow.BibliotecaRepository.ObterPorUsuarioIdAsync(evento.UsuarioId);
            if (biblioteca is null)
            {
                biblioteca = new Biblioteca
                {
                    Id = Guid.NewGuid(),
                    UsuarioId = evento.UsuarioId,
                    CriadaEm = DateTime.UtcNow
                };
                _uow.BibliotecaRepository.Adicionar(biblioteca);
                await _uow.CommitAsync(context.CancellationToken);
            }

            var item = new ItemBiblioteca
            {
                Id = Guid.NewGuid(),
                BibliotecaId = biblioteca.Id,
                JogoId = evento.JogoId,
                DataAdicao = DateTime.UtcNow
            };
            _uow.BibliotecaRepository.AdicionarItem(item);

            _logger.LogInformation(
                "Jogo {JogoId} adicionado à biblioteca do usuário {UsuarioId}.",
                evento.JogoId, evento.UsuarioId);
        }
        else
        {
            pedido.Status = StatusPedido.Rejeitado;
            _uow.PedidoRepository.Atualizar(pedido);

            _logger.LogInformation(
                "Pedido {PedidoId} rejeitado. Motivo={Motivo}",
                pedido.Id, evento.Motivo);
        }

        await _uow.CommitAsync(context.CancellationToken);
    }
}

**Consome:** `UsuarioCriadoEvento`, `PagamentoProcessadoEvento`

---

## Endpoints

| Método | Rota | Auth | Descrição |
|--------|------|------|-----------|
| GET | `/jogos` | Bearer (qualquer) | Lista catálogo de jogos |
| POST | `/jogos` | Bearer (Admin) | Cria novo jogo |
| PUT | `/jogos/{id}` | Bearer (Admin) | Atualiza jogo |
| DELETE | `/jogos/{id}` | Bearer (Admin) | Remove jogo |
| POST | `/compras` | Bearer (qualquer) | Inicia compra de jogo (assíncrono) |
| GET | `/compras/{pedidoId}` | Bearer (qualquer) | Consulta status do pedido |
| GET | `/biblioteca/{usuarioId}` | Bearer (qualquer) | Lista jogos do usuário |
| GET | `/health` | Sem auth | Health check |

JWT é emitido **exclusivamente** pelo UsersAPI (`:5001`). Este serviço apenas valida o token.
O `UsuarioId` é extraído do claim `sub` do JWT — não vem no body das requisições.

| Evento | Quando |
|--------|--------|

### Consumidos
| Evento | Ação |
|--------|------|

## Como rodar com Docker

```bash
# Build
docker build --build-arg NUGET_AUTH_TOKEN=$NUGET_AUTH_TOKEN -t fcg-catalog-api .

# Run
docker run -d \
  -p 5002:8080 \
  -e ConnectionStrings__DefaultConnection="Data Source=/data/catalog.db" \
  -e Jwt__Key="<mesma-chave-do-users-api>" \
  -e Jwt__Issuer="AppFiapFcGames" \
  -e RabbitMQ__Host="host.docker.internal" \
  -e RabbitMQ__Username="guest" \
  -e RabbitMQ__Password="guest" \
  -v catalog-data:/data \
  fcg-catalog-api
